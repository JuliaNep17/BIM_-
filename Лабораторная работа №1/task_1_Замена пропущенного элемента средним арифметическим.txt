numbers = [2, -93, -2, 8, None, -44, -1, -85, -14, 90, -22, -90, -100, -8, 38, -92, -45, 67, 53, 25]

sum_of_numbers = sum(filter(lambda x: isinstance(x, (int, float)), numbers))
count_of_elements = len(numbers)
average_of_numbers = sum_of_numbers / count_of_elements

index = numbers.index(None)
numbers[index] = average_of_numbers

print("Измененный список:", numbers)
